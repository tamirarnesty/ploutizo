### Task 2: Create ErrorBoundary component

**Type:** create

**File:** `apps/web/src/components/error-boundary/ErrorBoundary.tsx`

**Action:**

Create the error boundary component for unhandled render errors. Receives `ErrorComponentProps` from `@tanstack/react-router`. Uses `useState` for the dev-mode toggle.

Exact implementation:

```tsx
import { useState } from 'react';
import { Link, ErrorComponentProps } from '@tanstack/react-router';
import { Button } from '@ploutizo/ui/components/button';
import { Text } from '@ploutizo/ui/components/text';

const ART_ERR = ` ███████╗██████╗ ██████╗
 ██╔════╝██╔══██╗██╔══██╗
 █████╗  ██████╔╝██████╔╝
 ██╔══╝  ██╔══██╗██╔══██╗
 ███████╗██║  ██║██║  ██║
 ╚══════╝╚═╝  ╚═╝╚═╝  ╚═╝`;

export const ErrorBoundary = ({ error, reset }: ErrorComponentProps) => {
  const [showDetails, setShowDetails] = useState(false);

  return (
    <div className="flex min-h-screen flex-col items-center justify-center bg-background px-6">
      <div className="w-full max-w-lg">
        {/* Status bar */}
        <div className="flex items-center justify-between border-b border-border pb-3 mb-8">
          <span className="font-mono text-xs tracking-widest text-muted-foreground">
            ● STATUS — ERROR
          </span>
          <span className="font-mono text-xs tracking-widest text-muted-foreground">
            ploutizo/~/error
          </span>
        </div>

        {/* Art block */}
        <pre className="font-mono text-xs leading-[1.15] text-muted-foreground opacity-40 select-none mb-6">
          {ART_ERR}
        </pre>

        {/* Error label */}
        <p className="font-mono text-xs tracking-widest text-muted-foreground mb-6">
          ERROR · 0x500
        </p>

        {/* Heading + subtext */}
        <div className="space-y-2 mb-8">
          <Text as="h1" variant="h1">
            Something went wrong
          </Text>
          <Text variant="body-sm" className="text-muted-foreground">
            An unexpected error occurred. Try again, or head back to the dashboard.
          </Text>
        </div>

        {/* Action buttons */}
        <div className="flex gap-2">
          <Button onClick={reset}>Try again</Button>
          <Link to="/dashboard">
            <Button variant="outline">Go to Dashboard</Button>
          </Link>
        </div>

        {/* Dev-mode error detail — not rendered in production */}
        {import.meta.env.DEV && (
          <div className="mt-6 space-y-2">
            <p className="font-mono text-xs text-muted-foreground">
              {error.message}
            </p>
            <Button
              variant="link"
              className="h-auto p-0 font-mono text-xs text-muted-foreground"
              onClick={() => setShowDetails((v) => !v)}
            >
              {showDetails ? 'Hide details' : 'Show details'}
            </Button>
            {showDetails && (
              <pre className="font-mono text-xs text-muted-foreground bg-muted rounded p-3 max-h-[200px] overflow-y-auto whitespace-pre-wrap">
                {error.stack}
              </pre>
            )}
          </div>
        )}
      </div>
    </div>
  );
};
```

Notes:

- `useState` toggle: collapsed by default (`false`), flip on button click
- The toggle `<button>` is a plain HTML button (no shadcn equivalent for an inline text toggle)
- Do NOT use `forwardRef` (React 19 project)
