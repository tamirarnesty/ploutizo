Step 4 — Create `packages/ui/src/components/theme-toggle.tsx`:

```tsx
"use client"

import { useTheme } from "next-themes"
import { Monitor as MonitorIcon, Sun as SunIcon, Moon as MoonIcon } from "lucide-react"
import { Button } from "@ploutizo/ui/components/button"
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from "@ploutizo/ui/components/tooltip"

const cycleMap = { system: "light", light: "dark", dark: "system" } as const
type Theme = keyof typeof cycleMap

export function ThemeToggle() {
  const { theme, setTheme } = useTheme()
  const current = (theme ?? "system") as Theme
  const label = current.charAt(0).toUpperCase() + current.slice(1)

  return (
    <TooltipProvider>
      <Tooltip>
        <TooltipTrigger asChild>
          <Button
            variant="ghost"
            size="icon"
            className="size-8"
            onClick={() => setTheme(cycleMap[current])}
          >
            {current === "system" && <MonitorIcon />}
            {current === "light" && <SunIcon />}
            {current === "dark" && <MoonIcon />}
            <span className="sr-only">{label}</span>
          </Button>
        </TooltipTrigger>
        <TooltipContent>{label}</TooltipContent>
      </Tooltip>
    </TooltipProvider>
  )
}
```

Key implementation details (per D-05, D-06, D-07, UI-SPEC):
- `theme ?? "system"` null-guard: next-themes returns `undefined` before client mount; fall back to "system" (matches `defaultTheme`)
- Cycle map: `{ system: "light", light: "dark", dark: "system" }` — exact order per D-05
- Icons: Monitor for system, Sun for light, Moon for dark — per D-06
- Button: `variant="ghost" size="icon" className="size-8"` — 32px touch target per UI-SPEC
- Tooltip: `TooltipProvider` is outermost (not inner) wrapper — CRITICAL: avoids silent Radix context failure
- Tooltip text = `label` (capitalized stored theme, e.g. "System") — per D-07
- `"use client"` directive: required because ThemeToggle uses `useTheme` hook (client-side state)
- No `dark:` utility overrides: `variant="ghost"` in sidebar context receives correct hover tokens via `bg-sidebar-accent`/`text-sidebar-accent-foreground` automatically

Modify `RootDocument` to add `suppressHydrationWarning` to `<html>` and wrap body content with `ThemeProvider`:

```tsx
const RootDocument = ({ children }: { children: React.ReactNode }) => (
  <html lang="en" suppressHydrationWarning>
    <head>
      <HeadContent />
    </head>
    <body>
      <ThemeProvider
        attribute="class"
        defaultTheme="system"
        enableSystem
        disableTransitionOnChange
        storageKey="theme"
      >
        <QueryClientProvider client={queryClient}>
          <ClerkProvider appearance={{ theme: shadcn }}>
            <TokenInitializer />
            {children}
          </ClerkProvider>
        </QueryClientProvider>
      </ThemeProvider>
      <Scripts />
    </body>
  </html>
)
```

Key details:
- `suppressHydrationWarning` on `<html>`: silences React's warning about server/client class attribute mismatch caused by next-themes adding `.dark` class on client (standard next-themes requirement, per RESEARCH.md Pitfall 2)
- `ThemeProvider` wraps `<body>` content (NOT `<html>`) — `ThemeProvider` renders a `<div>` wrapper which cannot parent `<html>` (per RESEARCH.md Pitfall 2)
- `ThemeProvider` is outermost content wrapper — inside `<body>`, outside `QueryClientProvider` and `ClerkProvider`
- `<Scripts />` stays outside `ThemeProvider` — it doesn't need theme context
- Props: `attribute="class"` (matches `@custom-variant dark` in globals.css), `defaultTheme="system"` (D-02), `enableSystem` (enables OS preference), `disableTransitionOnChange` (prevents color flash, per RESEARCH.md Pitfall 4), `storageKey="theme"` (D-03, next-themes default)
- `ClerkProvider appearance={{ theme: shadcn }}` stays UNCHANGED (per D-08 — shadcn Clerk theme auto-adapts to `.dark` class; no dynamic switching needed)
