### `hooks/useTransactionForm.ts` (hook, request-response) — NO ANALOG

**No analog.** AccountForm inlines `useAppForm` inside `AccountFormInner`. D-24 prescribes extraction to a dedicated hook file. This is a new repo pattern.

**Pattern to follow — RESEARCH.md Pattern 3:**
```ts
// apps/web/src/components/transactions/hooks/useTransactionForm.ts
export function useTransactionForm(transaction: TransactionRow | null, onClose: () => void) {
  const isEditing = transaction !== null
  const createTransaction = useCreateTransaction()
  const updateTransaction = useUpdateTransaction(transaction?.id ?? "")

  const form = useAppForm({
    defaultValues: buildDefaultValues(transaction),  // factory function
    onSubmit: async ({ value }) => {
      const payload = toApiPayload(value)  // dollars→cents + strip undefined type-specifics
      const mutation = isEditing ? updateTransaction : createTransaction
      mutation.mutate(payload, {
        onSuccess: onClose,
        onError: () =>
          form.setErrorMap({
            onSubmit: "Couldn't save changes. Check your connection and try again.",
          }),
      })
    },
  })

  return { form, isEditing }
}
```

**`buildDefaultValues` factory:**
```ts
function buildDefaultValues(transaction: TransactionRow | null): TransactionFormValues {
  return {
    type: transaction?.type ?? 'expense',
    accountId: transaction?.accountId ?? '',
    amount: transaction ? transaction.amount / 100 : 0,  // cents → dollars
    date: transaction?.date ?? new Date().toISOString().slice(0, 10),
    description: transaction?.description ?? undefined,
    merchant: transaction?.merchant ?? undefined,
    assignees: transaction?.assignees.map(toAssigneeFormRow) ?? [],
    tagIds: transaction?.tags.map((t) => t.id) ?? [],
    // Type-specific — undefined if not relevant to type
    categoryId: transaction?.categoryId ?? undefined,
    refundOf: transaction?.refundOf ?? undefined,
    incomeType: transaction?.incomeType ?? undefined,
    incomeSource: transaction?.incomeSource ?? undefined,
    toAccountId: transaction?.toAccountId ?? undefined,
    settledAccountId: transaction?.settledAccountId ?? undefined,
    investmentType: transaction?.investmentType ?? undefined,
  }
}
```

### `apps/web/src/lib/lrm.ts` (utility, transform) — NO ANALOG

**No analog.** New pure function file. Use RESEARCH.md Pattern 5 directly.

```ts
// apps/web/src/lib/lrm.ts
export function lrmSplit(
  totalCents: number,
  memberIds: string[]
): { memberId: string; amountCents: number; percentage: number }[] {
  if (memberIds.length === 0) return []
  const base = Math.floor(totalCents / memberIds.length)
  const remainder = totalCents - base * memberIds.length
  const exactPct = 100 / memberIds.length
  return memberIds.map((memberId, i) => ({
    memberId,
    amountCents: i < remainder ? base + 1 : base,
    percentage: parseFloat(exactPct.toFixed(3)),
  }))
}
```

**Testable:** No React imports, no DOM dependencies. Add vitest to `apps/web` (Wave 0) and test at `apps/web/src/lib/lrm.test.ts`.

### `useGetTransaction.ts` (query hook, single entity) — ROLE-MATCH ANALOG

export const fetchTransaction = async (id: string): Promise<TransactionRow> => {

export const useGetTransaction = (id: string | null) => {
