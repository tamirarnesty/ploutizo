      const { data: overview, isLoading: overviewLoading } = useGetHouseholdOverview()

        defaultValues: { email: "" },
        validators: {
          onSubmit: ({ value }) => {
            const r = z.object({ email: z.string().email("Enter a valid email address") }).safeParse(value)
            if (!r.success) return r.error.issues.map((i) => i.message).join(", ")
          },
        },
        onSubmit: ({ value }) => {
          inviteMutation.mutate(value.email, {
            onSuccess: () => {
              form.reset()
              toast.success(`Invite sent to ${value.email}.`)
            },
            onError: (err) => {
              const code = (err as { error?: { code?: string } })?.error?.code
              if (code === "ALREADY_MEMBER") toast.error("Already a member of this household.")
              else if (code === "INVITATION_PENDING") toast.error("An invitation is already pending for this email.")
              else toast.error("Failed to send invite. Try again.")

          })

| T-03.2.1-03-03 | Denial of Service | Invite form — rapid submission | accept | Button disabled during in-flight mutation (`isSubmitting`). No additional rate limiting in v1; Clerk's API rate limits apply server-side. |
