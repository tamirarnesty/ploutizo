```typescript
const clerkRes = await fetch(
```

```typescript
// D-11: Use status field from Clerk — NOT date math for expired state
function getExpiryInfo(expiresAt: string | null, status: PendingInvitation['status']) {
  if (status === 'expired') {
    return { label: 'Expired', className: 'text-destructive' }
  }
  if (!expiresAt) return null
  const daysLeft = Math.ceil(
    (new Date(expiresAt).getTime() - Date.now()) / (1000 * 60 * 60 * 24)
  )
  const label = `Expires in ${daysLeft} day${daysLeft === 1 ? '' : 's'}`
  const className = daysLeft <= 2 ? 'text-warning-foreground' : 'text-muted-foreground'
  return { label, className }
}
```
