```typescript
transactionsRouter.patch('/:id', async (c) => {
  const { orgId } = getAuth(c)
  const id = c.req.param('id')
  const result = updateTransactionSchema.safeParse(await c.req.json())
  if (!result.success) {
    return c.json({ error: { code: 'VALIDATION_ERROR', errors: result.error.issues } }, 400)
  }
  const data = result.data

  // D-11: re-validate split sum if assignees provided
  if (data.assignees && data.assignees.length > 0 && data.amount !== undefined) {
    const sum = data.assignees.reduce((acc, a) => acc + a.amountCents, 0)
    if (sum !== data.amount) {
      return badRequest(c, 'Assignee amounts must sum to transaction amount')
    }
  }

  const { assignees, tagIds, ...updateData } = data

  const updated = await db.transaction(async (tx) => {
    // Pitfall 7: all three conditions — id, orgId, isNull(deletedAt)
    const rows = await tx
      .update(transactions)
      .set({ ...(updateData as Record<string, unknown>), updatedAt: new Date() })
      .where(and(
        eq(transactions.id, id),
        eq(transactions.orgId, orgId!),
        isNull(transactions.deletedAt),
      ))
      .returning()

    if (!rows.length) return null

    // D-03: replace-all assignees if provided in payload
    if (assignees !== undefined) {
      await tx.delete(transactionAssignees).where(eq(transactionAssignees.transactionId, id))
      if (assignees.length > 0) {
        await tx.insert(transactionAssignees).values(
          assignees.map((a) => ({
            transactionId: id,
            memberId: a.memberId,
            amountCents: a.amountCents,
            percentage: a.percentage != null ? a.percentage.toString() : null,
          }))
        )
      }
    }

    // Replace-all tags if provided in payload
    if (tagIds !== undefined) {
      await tx.delete(transactionTags).where(eq(transactionTags.transactionId, id))
      if (tagIds.length > 0) {
        await tx.insert(transactionTags).values(
          tagIds.map((tagId) => ({ transactionId: id, tagId }))
        )
      }
    }

    return rows[0]
  })

  if (!updated) {
    return c.json({ error: { code: 'NOT_FOUND', message: 'Transaction not found.' } }, 404)
  }

  return c.json({ data: updated })
})
```

```typescript
transactionsRouter.delete('/:id', async (c) => {
  const { orgId } = getAuth(c)
  const id = c.req.param('id')

  // D-15: soft delete — set deletedAt, never hard delete
  const [updated] = await db
    .update(transactions)
    .set({ deletedAt: new Date(), updatedAt: new Date() })
    .where(and(
      eq(transactions.id, id),
      eq(transactions.orgId, orgId!),
      isNull(transactions.deletedAt),  // idempotency: already-deleted → 404
    ))
    .returning()

  if (!updated) {
    return c.json({ error: { code: 'NOT_FOUND', message: 'Transaction not found.' } }, 404)
  }

  return c.json({ data: { id: updated.id } })
})
```

<threat_model>
| Threat | Severity | Mitigation |
|--------|----------|------------|
| Tenant isolation bypass on PATCH/DELETE | HIGH | PATCH uses `and(eq(transactions.id, id), eq(transactions.orgId, orgId!), isNull(transactions.deletedAt))` in update WHERE — a cross-org id returns 0 rows → 404; orgId from Clerk JWT only |
| Soft-delete bypass on PATCH (editing deleted transaction) | HIGH | `isNull(transactions.deletedAt)` in PATCH WHERE clause — deleted transactions return 0 rows → 404 |
| Double-delete / idempotency | MED | DELETE WHERE includes `isNull(transactions.deletedAt)` — second DELETE returns 404 (acceptable: already deleted) |
| Replace-all assignees partial failure | MED | PATCH assignees + tags replace-all runs inside `db.transaction()` — any failure auto-rolls back; no orphaned partial state |
| PATCH type mutation | MED | `updateTransactionSchema` (from validators) does not include `type` field — type is structurally excluded from updates; Drizzle only sets columns present in `updateData` |
| 404 oracle (leaking whether id exists vs wrong org) | LOW | All PATCH/DELETE WHERE clauses combine id + orgId + deletedAt — always 404 on any mismatch; never a distinguishable 403 |
</threat_model>
