    ```typescript
    // PATCH: update scalar fields + replace-all assignees/tags in a single DB transaction
    // Returns updated row or null (not found / wrong org / already deleted — Pitfall 7)
    export async function updateTransaction(
      id: string,
      orgId: string,
      data: z.infer<typeof updateTransactionSchema>
    ) {
      // D-11: re-validate split sum if both assignees and amount are present in patch
      if (data.assignees && data.assignees.length > 0 && data.amount !== undefined) {
        const splitError = validateSplitSum(data.amount, data.assignees)
        if (splitError) throw new Error(splitError)
      }

      const { assignees, tagIds, ...updateData } = data

      return db.transaction(async (tx) => {
        // Pitfall 7: three-condition WHERE — id + orgId + isNull(deletedAt)
        const rows = await tx
          .update(transactions)
          .set({ ...(updateData as Record<string, unknown>), updatedAt: new Date() })
          .where(and(
            eq(transactions.id, id),
            eq(transactions.orgId, orgId),
            isNull(transactions.deletedAt),
          ))
          .returning()

        if (!rows.length) return null

        // D-03: replace-all assignees if provided in payload
        if (assignees !== undefined) {
          await tx.delete(transactionAssignees)
            .where(eq(transactionAssignees.transactionId, id))
          if (assignees.length > 0) {
            await tx.insert(transactionAssignees).values(
              assignees.map((a) => ({
                transactionId: id,
                memberId: a.memberId,
                amountCents: a.amountCents,
                percentage: a.percentage != null ? a.percentage.toString() : null,
              }))
            )
          }
        }

        // Replace-all tags if provided
        if (tagIds !== undefined) {
          await tx.delete(transactionTags)
            .where(eq(transactionTags.transactionId, id))
          if (tagIds.length > 0) {
            await tx.insert(transactionTags).values(
              tagIds.map((tagId) => ({ transactionId: id, tagId }))
            )
          }
        }

        return rows[0]
      })
    }

    // DELETE: soft-delete by setting deletedAt = now() (D-15)
    // Returns {id} on success or null if not found/wrong org/already deleted
    export async function deleteTransaction(
      id: string,
      orgId: string
    ): Promise<{ id: string } | null> {
      const [updated] = await db
        .update(transactions)
        .set({ deletedAt: new Date(), updatedAt: new Date() })
        .where(and(
          eq(transactions.id, id),
          eq(transactions.orgId, orgId),
          isNull(transactions.deletedAt),  // idempotency: already-deleted → 0 rows → null
        ))
        .returning({ id: transactions.id })

      return updated ?? null
    }
    ```

    Note: updateTransaction throws Error with the split error message when sum validation fails.
    The PATCH route handler catches this and returns badRequest(). This keeps the service layer
    free of HTTP concerns while still surfacing validation errors.

    Alternative if throw feels wrong: return { error: string } | row. Choose the throw approach
    — it matches how Drizzle db.transaction() propagates errors (auto-rollback on throw).

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03.2-08 | Information Disclosure | PATCH tenant isolation | mitigate | updateTransaction WHERE includes eq(transactions.orgId, orgId) — cross-org id returns null → 404; orgId from Clerk JWT only |
| T-03.2-09 | Information Disclosure | DELETE tenant isolation | mitigate | deleteTransaction WHERE includes eq(transactions.orgId, orgId) — same pattern; 404 on mismatch |
| T-03.2-10 | Information Disclosure | 404 oracle on PATCH/DELETE | accept | Both return 404 for missing/wrong-org/already-deleted without distinguishing; no 403 leakage (Pitfall 7) |
| T-03.2-11 | Tampering | PATCH soft-delete bypass | mitigate | updateTransaction WHERE includes isNull(transactions.deletedAt) — deleted rows return 0 rows → null → 404 |
| T-03.2-12 | Tampering | DELETE double-delete | mitigate | deleteTransaction WHERE includes isNull(transactions.deletedAt) — second DELETE returns null → 404 |
| T-03.2-13 | Tampering | PATCH type mutation | mitigate | updateTransactionSchema structurally excludes 'type' field — type cannot be changed post-creation |
| T-03.2-14 | Tampering | PATCH replace-all partial failure | mitigate | delete + insert inside db.transaction() — any error auto-rolls back; no orphaned partial state |
| T-03.2-15 | Tampering | PATCH split sum on partial patch | mitigate | Sum validation only fires when BOTH assignees AND amount are present in payload; partial patches without amount skip validation correctly |
