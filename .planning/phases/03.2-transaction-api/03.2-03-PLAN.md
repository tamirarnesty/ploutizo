    ```typescript
    // PATCH: update scalar fields + replace-all assignees/tags in a single DB transaction
    // Returns updated row or null (not found / wrong org / already deleted — Pitfall 7)
    export async function updateTransaction(
      id: string,
      orgId: string,
      data: z.infer<typeof updateTransactionSchema>
    ) {
      // D-11: re-validate split sum if both assignees and amount are present in patch
      if (data.assignees && data.assignees.length > 0 && data.amount !== undefined) {
        const splitError = validateSplitSum(data.amount, data.assignees)
        if (splitError) throw new Error(splitError)
      }

      const { assignees, tagIds, ...updateData } = data

      return db.transaction(async (tx) => {
        // Delegate scalar UPDATE to query layer (Pitfall 7: three-condition WHERE)
        const updated = await updateTransactionScalarsQuery(
          tx,
          id,
          orgId,
          updateData as Record<string, unknown>
        )

        if (!updated) return null

        // D-03: replace-all assignees if provided in payload (multi-table coordination — stays here)
        if (assignees !== undefined) {
          await tx.delete(transactionAssignees)
            .where(eq(transactionAssignees.transactionId, id))
          if (assignees.length > 0) {
            await tx.insert(transactionAssignees).values(
              assignees.map((a) => ({
                transactionId: id,
                memberId: a.memberId,
                amountCents: a.amountCents,
                percentage: a.percentage != null ? a.percentage.toString() : null,
              }))
            )
          }
        }

        // Replace-all tags if provided (multi-table coordination — stays here)
        if (tagIds !== undefined) {
          await tx.delete(transactionTags)
            .where(eq(transactionTags.transactionId, id))
          if (tagIds.length > 0) {
            await tx.insert(transactionTags).values(
              tagIds.map((tagId) => ({ transactionId: id, tagId }))
            )
          }
        }

        return updated
      })
    }

    // DELETE: soft-delete — delegates to query layer (D-15)
    // Returns {id} on success or null if not found/wrong org/already deleted
    export async function deleteTransaction(
      id: string,
      orgId: string
    ): Promise<{ id: string } | null> {
      return softDeleteTransactionQuery(id, orgId)
    }
    ```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |

| T-03.2-08 | Information Disclosure | PATCH tenant isolation | mitigate | updateTransactionScalarsQuery WHERE includes eq(transactions.orgId, orgId) — cross-org id returns null → 404; orgId from Clerk JWT only |
| T-03.2-09 | Information Disclosure | DELETE tenant isolation | mitigate | softDeleteTransactionQuery WHERE includes eq(transactions.orgId, orgId) — same pattern; 404 on mismatch |
| T-03.2-10 | Information Disclosure | 404 oracle on PATCH/DELETE | accept | Both return 404 for missing/wrong-org/already-deleted without distinguishing; no 403 leakage (Pitfall 7) |
| T-03.2-11 | Tampering | PATCH soft-delete bypass | mitigate | updateTransactionScalarsQuery WHERE includes isNull(transactions.deletedAt) — deleted rows return 0 rows → null → 404 |
| T-03.2-12 | Tampering | DELETE double-delete | mitigate | softDeleteTransactionQuery WHERE includes isNull(transactions.deletedAt) — second DELETE returns null → 404 |
| T-03.2-13 | Tampering | PATCH type mutation | mitigate | updateTransactionSchema structurally excludes 'type' field — type cannot be changed post-creation |
| T-03.2-14 | Tampering | PATCH replace-all partial failure | mitigate | delete + insert inside db.transaction() — any error auto-rolls back; no orphaned partial state |
