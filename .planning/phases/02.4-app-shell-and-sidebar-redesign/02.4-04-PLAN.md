Rewrite the file:

```tsx
import { createFileRoute, Outlet, useNavigate, useRouterState } from "@tanstack/react-router"
import { Tabs, TabsList, TabsTrigger } from "@ploutizo/ui/components/tabs"

export const Route = createFileRoute("/_layout/settings")({
  component: SettingsLayout,
})

const settingsTabs = [
  { label: "Categories & Tags", value: "/settings/categories" },
  { label: "Merchant Rules", value: "/settings/merchant-rules" },
  { label: "Household", value: "/settings/household" },
] as const

function SettingsLayout() {
  const { location } = useRouterState()
  const navigate = useNavigate()

  return (
    <div className="flex flex-col gap-6">
      <Tabs
        value={location.pathname}
        onValueChange={(value) => navigate({ to: value as string })}
      >
        <TabsList>
          {settingsTabs.map(({ label, value }) => (
            <TabsTrigger key={value} value={value}>
              {label}
            </TabsTrigger>
          ))}
        </TabsList>
      </Tabs>
      <Outlet />
    </div>
  )
}
```

Key points:
- `Tabs value={location.pathname}` — active tab driven by URL, not local state
- `onValueChange` calls `navigate({ to: value })` — TanStack Router programmatic navigation
- No `TabsContent` — `<Outlet />` renders the active route's component
- `TabsList` uses default shadcn pill appearance — no className overrides
- `value` in `settingsTabs` is the full path string (matches `location.pathname` exactly)
- No icons on tabs (shadcn Tabs don't typically have icons in the trigger)
- No custom vertical nav layout — outer div is `flex flex-col gap-6`
- No `p-6` on outer div (shell's `<main className="p-6">` handles padding)
