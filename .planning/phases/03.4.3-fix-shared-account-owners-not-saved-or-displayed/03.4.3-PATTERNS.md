```typescript
// GET / enrichment — fetch member display names for a set of account IDs.
// Guard against inArray([]) which generates invalid SQL (Pitfall 4 in RESEARCH.md).
export async function listAccountMemberDetails(accountIds: string[]) {
  if (accountIds.length === 0) return [];
  return db
    .select({
      accountId: accountMembers.accountId,
      memberId: orgMembers.id,
      displayName: orgMembers.displayName,
    })
    .from(accountMembers)
    .innerJoin(orgMembers, eq(orgMembers.id, accountMembers.memberId))
    .where(inArray(accountMembers.accountId, accountIds));
}
```

Note on join condition: `accountMembers.memberId` is a FK to `orgMembers.id` — join must be `eq(orgMembers.id, accountMembers.memberId)`, matching how `enrichTransactions` does `eq(transactionAssignees.memberId, orgMembers.id)`.

New `listAccounts` implementation in services/accounts.ts (replace lines 16–18):
```typescript
export async function listAccounts(orgId: string, includeArchived: boolean) {
  const rows = await listAccountsQuery(orgId, includeArchived);
  if (rows.length === 0) return [];
  const memberRows = await listAccountMemberDetails(rows.map((r) => r.id));
  // Map for O(1) grouping (js-perf skill rule)
  const byAccount = new Map<string, { id: string; displayName: string }[]>();
  for (const m of memberRows) {
    const list = byAccount.get(m.accountId) ?? [];
    list.push({ id: m.memberId, displayName: m.displayName });
    byAccount.set(m.accountId, list);
  }
  return rows.map((r) => ({ ...r, owners: byAccount.get(r.id) ?? [] }));
}
```

Note: uses `Map` not `Record` for O(1) grouping, per js-perf skill rule. Uses sequential queries (not `Promise.all`) since the second depends on the first's IDs.

## Shared Patterns

### Two-Query Merge (enrich pattern)
**Source:** `apps/api/src/lib/queries/transactions.ts` lines 340–386 (`enrichTransactions`)
**Apply to:** `listAccountMemberDetails` in `apps/api/src/lib/queries/accounts.ts`

Key invariants to copy:
1. Guard `if (ids.length === 0) return []` before `inArray` (Pitfall 4 / transactions Pitfall 6)
2. `db.select({ ... specific fields ... })` — never `db.select()` for join queries (avoids column name collisions)
3. `innerJoin` for required join (member must exist in org_members)
4. `inArray(fkColumn, parentIds)` to scope to fetched parent set
