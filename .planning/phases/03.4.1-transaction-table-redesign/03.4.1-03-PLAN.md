```typescript
// Validate counterpartAccountId belongs to the same org — T1 security mitigation
export async function counterpartAccountBelongsToOrg(
  accountId: string,
  orgId: string
): Promise<boolean> {
  const [row] = await db
    .select({ id: accounts.id })
    .from(accounts)
    .where(and(eq(accounts.id, accountId), eq(accounts.orgId, orgId)))
    .limit(1)
  return !!row
}
```

```typescript
}
```
