interface DescriptionSyncerProps {

Add this block between the destructure and the `return db.transaction(...)` call:

```typescript
const { assignees, tagIds, ...updateData } = data

// WR-01: when type changes, explicitly null FK columns that do not apply to the new type.
// Without this, switching from transfer to expense leaves counterpartAccountId in the DB
// and the table column renders "A → B" for what is now an expense row.
if (data.type) {
  const typeSpecificNulls: Record<string, null> = {}
  // counterpartAccountId applies only to transfer and settlement
  if (!['transfer', 'settlement'].includes(data.type)) {
    typeSpecificNulls.counterpartAccountId = null
  }
  // refundOf applies only to refund type
  if (data.type !== 'refund') {
    typeSpecificNulls.refundOf = null
  }
  // incomeType applies only to income type (if column exists in schema — verify)
  if (data.type !== 'income') {
    typeSpecificNulls.incomeType = null
  }
  // categoryId applies only to expense and refund
  if (!['expense', 'refund'].includes(data.type)) {
    typeSpecificNulls.categoryId = null
  }
  // Merge nulls into updateData — updateTransactionScalarsQuery will SET these to NULL
  Object.assign(updateData, typeSpecificNulls)
}
```
