Write this exact YAML content:

```yaml
name: CI

on:
  pull_request:
  push:
    branches:
      - main

concurrency:
  group: ci-${{ github.ref }}
  cancel-in-progress: true

permissions:
  contents: read

jobs:
  lint-typecheck:
    name: Lint & Typecheck
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v6

      - name: Setup pnpm
        uses: pnpm/action-setup@v6
        with:
          version: 9.15.9

      - name: Setup Node
        uses: actions/setup-node@v6
        with:
          node-version: 22.14.0
          cache: pnpm

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Lint
        run: pnpm turbo lint

      - name: Typecheck
        run: pnpm turbo typecheck

  test:
    name: Tests
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v6

      - name: Setup pnpm
        uses: pnpm/action-setup@v6
        with:
          version: 9.15.9

      - name: Setup Node
        uses: actions/setup-node@v6
        with:
          node-version: 22.14.0
          cache: pnpm

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Test
        run: pnpm turbo test

  format-check:
    name: Format Check
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v6

      - name: Setup pnpm
        uses: pnpm/action-setup@v6
        with:
          version: 9.15.9

      - name: Setup Node
        uses: actions/setup-node@v6
        with:
          node-version: 22.14.0
          cache: pnpm

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Format check
        run: pnpm turbo format:check

  build:
    name: Build
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v6

      - name: Setup pnpm
        uses: pnpm/action-setup@v6
        with:
          version: 9.15.9

      - name: Setup Node
        uses: actions/setup-node@v6
        with:
          node-version: 22.14.0
          cache: pnpm

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Build
        run: pnpm turbo build
```

Key design decisions embedded above:
- `pnpm/action-setup@v6` — latest major (v6.0.3); `version: 9.15.9` matches packageManager field
- `actions/setup-node@v6` with `cache: pnpm` — caches the pnpm store keyed to pnpm-lock.yaml
- `--frozen-lockfile` — fails if lockfile is out of date (catches missing pnpm install commits)
- Lint and typecheck share one job (D-02 says "lint + typecheck" as one concurrent unit) to save runner minutes while still being parallel with the other three jobs
- No `TURBO_TOKEN` or remote cache env vars (per Claude's Discretion: skip Turbo remote cache)
- No secrets or env vars needed (all tests are mocked, no live DB)
- `concurrency.cancel-in-progress: true` — cancels stale runs when new commit pushed to same PR branch
- `permissions: contents: read` — least-privilege; workflow only needs to read the repo

| Boundary | Description |
|----------|-------------|
| PR author → GitHub Actions | Untrusted code runs in workflow on every PR — must not have write permissions |
| runner → pnpm cache | Cache poisoning risk if cache key is predictable — mitigated by `actions/setup-node` built-in pnpm cache (keyed to lockfile hash) |
